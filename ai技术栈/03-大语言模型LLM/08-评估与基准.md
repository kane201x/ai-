### 7.1 Perplexity 计算

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

@torch.no_grad()
def compute_perplexity(model, input_ids, stride=512, max_len=4096):
    model.eval()
    nlls = []
    seq_len = input_ids.size(1)
    prev_end = 0
    for begin in range(0, seq_len, stride):
        end = min(begin + max_len, seq_len)
        input_chunk = input_ids[:, begin:end]
        target_chunk = input_chunk.clone()
        logits = model(input_chunk).logits
        shift_logits = logits[:, :-1, :].contiguous()
        shift_labels = target_chunk[:, 1:].contiguous()
        loss = F.cross_entropy(shift_logits.view(-1, shift_logits.size(-1)), shift_labels.view(-1), reduction="none")
        nlls.append(loss)
        prev_end = end
    nlls = torch.cat(nlls)
    ppl = torch.exp(nlls.mean())
    return ppl.item()

def compute_perplexity_batched(model, dataset, batch_size=4, max_len=2048):
    model.eval()
    total_nll = 0.0
    total_tokens = 0
    with torch.no_grad():
        for batch in dataset:
            input_ids = batch["input_ids"][:, :max_len]
            logits = model(input_ids).logits
            shift_logits = logits[:, :-1, :].contiguous()
            shift_labels = input_ids[:, 1:].contiguous()
            loss = F.cross_entropy(shift_logits.view(-1, shift_logits.size(-1)), shift_labels.view(-1), reduction="sum")
            total_nll += loss.item()
            total_tokens += (shift_labels != -100).sum().item()
    return torch.exp(torch.tensor(total_nll / total_tokens)).item()
```

### 7.2 BLEU 计算

```python
from collections import Counter
import math

def compute_bleu(reference: str, hypothesis: str, max_n=4):
    ref_tokens = reference.split()
    hyp_tokens = hypothesis.split()

    def get_ngrams(tokens, n):
        return [tuple(tokens[i:i+n]) for i in range(len(tokens) - n + 1)]

    precisions = []
    for n in range(1, max_n + 1):
        ref_ngrams = Counter(get_ngrams(ref_tokens, n))
        hyp_ngrams = Counter(get_ngrams(hyp_tokens, n))
        matches = sum(min(hyp_ngrams[ng], ref_ngrams[ng]) for ng in hyp_ngrams)
        total = max(sum(hyp_ngrams.values()), 1)
        precisions.append(matches / total)

    if min(precisions) == 0:
        return 0.0

    bp = min(1.0, math.exp(1 - len(ref_tokens) / max(len(hyp_tokens), 1)))
    bleu = bp * math.exp(sum(math.log(p) for p in precisions) / max_n)
    return bleu

def corpus_bleu(references: List[str], hypotheses: List[str]):
    total_bleu = 0
    for ref, hyp in zip(references, hypotheses):
        total_bleu += compute_bleu(ref, hyp)
    return total_bleu / len(references)
```

### 7.3 ROUGE 计算

```python
def compute_rouge_n(reference: str, hypothesis: str, n=1):
    ref_ngrams = Counter([tuple(reference.split()[i:i+n]) for i in range(len(reference.split()) - n + 1)])
    hyp_ngrams = Counter([tuple(hypothesis.split()[i:i+n]) for i in range(len(hypothesis.split()) - n + 1)])

    matches = sum(min(ref_ngrams[ng], hyp_ngrams[ng]) for ng in hyp_ngrams)
    ref_count = max(sum(ref_ngrams.values()), 1)
    hyp_count = max(sum(hyp_ngrams.values()), 1)

    precision = matches / hyp_count
    recall = matches / ref_count
    f1 = 2 * precision * recall / (precision + recall + 1e-8)
    return {"precision": precision, "recall": recall, "f1": f1}

def compute_rouge_l(reference: str, hypothesis: str):
    ref_words = reference.split()
    hyp_words = hypothesis.split()
    m, n = len(ref_words), len(hyp_words)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if ref_words[i-1] == hyp_words[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    lcs = dp[m][n]
    precision = lcs / max(n, 1)
    recall = lcs / max(m, 1)
    f1 = 2 * precision * recall / (precision + recall + 1e-8)
    return {"precision": precision, "recall": recall, "f1": f1}
```

### 7.4 MMLU 评估实现

```python
class MMLUEvaluator:
    def __init__(self, model, tokenizer):
        self.model = model
        self.tokenizer = tokenizer

    def format_question(self, question: str, choices: List[str]) -> str:
        choice_text = "\n".join([f"{chr(65+i)}. {c}" for i, c in enumerate(choices)])
        return f"问题: {question}\n选项:\n{choice_text}\n请直接输出选项字母 (A/B/C/D)。"

    @torch.no_grad()
    def evaluate_single(self, question: str, choices: List[str], answer: str) -> bool:
        prompt = self.format_question(question, choices)
        inputs = self.tokenizer(prompt, return_tensors="pt")
        logits = self.model(**inputs).logits[:, -1]
        option_ids = [self.tokenizer.encode(f" {chr(65+i)}")[-1] for i in range(len(choices))]
        option_probs = F.softmax(logits[0, option_ids], dim=-1)
        predicted = chr(65 + option_probs.argmax().item())
        return predicted == answer

    def evaluate(self, dataset: List[Dict]) -> Dict:
        correct = 0
        total = 0
        results_by_subject = {}
        for item in dataset:
            is_correct = self.evaluate_single(item["question"], item["choices"], item["answer"])
            correct += int(is_correct)
            total += 1
            subject = item.get("subject", "unknown")
            if subject not in results_by_subject:
                results_by_subject[subject] = {"correct": 0, "total": 0}
            results_by_subject[subject]["correct"] += int(is_correct)
            results_by_subject[subject]["total"] += 1
        accuracy = correct / max(total, 1)
        subject_acc = {s: v["correct"]/v["total"] for s, v in results_by_subject.items()}
        return {"accuracy": accuracy, "total": total, "by_subject": subject_acc}
```

### 7.5 LLM-as-Judge 实现

```python
class LLMAsJudge:
    def __init__(self, judge_model, judge_tokenizer):
        self.model = judge_model
        self.tokenizer = judge_tokenizer

    def single_score(self, question: str, answer: str, criteria: str = "准确性、完整性、相关性") -> float:
        prompt = f"""请根据以下标准评估回答质量。

问题: {question}
回答: {answer}

评估标准: {criteria}

请按 1-5 分评分（5 分最佳），只输出数字评分。"""
        inputs = self.tokenizer(prompt, return_tensors="pt")
        outputs = self.model.generate(**inputs, max_new_tokens=16)
        response = self.tokenizer.decode(outputs[0])
        score = int(''.join(filter(str.isdigit, response)) or 3)
        return min(max(score, 1), 5)

    def pairwise_compare(self, question: str, answer_a: str, answer_b: str) -> str:
        prompt = f"""请比较以下两个回答哪个更好。

问题: {question}
回答 A: {answer_a}
回答 B: {answer_b}

哪个回答更好？请输出 "A" 或 "B"。"""
        inputs = self.tokenizer(prompt, return_tensors="pt")
        outputs = self.model.generate(**inputs, max_new_tokens=16)
        response = self.tokenizer.decode(outputs[0]).upper()
        return "A" if "A" in response else "B"

    def g_eval(self, question: str, answer: str, criteria: str) -> Dict:
        prompt = f"""任务: 评估回答质量。

{criteria}

问题: {question}
回答: {answer}

请逐步推理，最后输出格式: [[评分: X]]"""
        inputs = self.tokenizer(prompt, return_tensors="pt")
        outputs = self.model.generate(**inputs, max_new_tokens=256)
        response = self.tokenizer.decode(outputs[0])
        import re
        match = re.search(r'评分:\s*(\d+(?:\.\d+)?)', response)
        score = float(match.group(1)) if match else 3.0
        return {"score": min(max(score, 1), 10), "reasoning": response}
```

### 7.6 评估参数采样与统计

```python
def evaluate_with_sampling(model, tokenizer, questions, num_samples=5, temperature=0.7):
    results = []
    for q in questions:
        samples = []
        for _ in range(num_samples):
            inputs = tokenizer(q, return_tensors="pt")
            outputs = model.generate(**inputs, max_new_tokens=256, temperature=temperature, do_sample=True)
            answer = tokenizer.decode(outputs[0], skip_special_tokens=True)
            samples.append(answer)
        results.append({"question": q, "samples": samples, "unique_answers": len(set(samples))})
    return results

def compute_pass_at_k(model, tokenizer, problems, k=5):
    correct = 0
    for prob in problems:
        for _ in range(k):
            inputs = tokenizer(prob["prompt"], return_tensors="pt")
            outputs = model.generate(**inputs, max_new_tokens=512, do_sample=True, temperature=0.8)
            answer = tokenizer.decode(outputs[0])
            if prob["test"](answer):
                correct += 1
                break
    return correct / len(problems)

def compute_confidence_intervals(scores, confidence=0.95):
    import numpy as np
    from scipy import stats
    scores = np.array(scores)
    mean = scores.mean()
    se = stats.sem(scores)
    ci = stats.t.interval(confidence, len(scores) - 1, loc=mean, scale=se)
    return {"mean": mean, "ci_lower": ci[0], "ci_upper": ci[1], "std": scores.std()}
```
