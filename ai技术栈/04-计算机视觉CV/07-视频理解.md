```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class C3D(nn.Module):
    def __init__(self, num_classes=400):
        super().__init__()
        self.conv1 = nn.Conv3d(3, 64, kernel_size=3, padding=1)
        self.pool1 = nn.MaxPool3d(2)
        self.conv2 = nn.Conv3d(64, 128, kernel_size=3, padding=1)
        self.pool2 = nn.MaxPool3d(2)
        self.conv3a = nn.Conv3d(128, 256, kernel_size=3, padding=1)
        self.conv3b = nn.Conv3d(256, 256, kernel_size=3, padding=1)
        self.pool3 = nn.MaxPool3d(2)
        self.conv4a = nn.Conv3d(256, 512, kernel_size=3, padding=1)
        self.conv4b = nn.Conv3d(512, 512, kernel_size=3, padding=1)
        self.pool4 = nn.MaxPool3d(2)
        self.conv5a = nn.Conv3d(512, 512, kernel_size=3, padding=1)
        self.conv5b = nn.Conv3d(512, 512, kernel_size=3, padding=1)
        self.pool5 = nn.MaxPool3d(2)
        self.fc6 = nn.Linear(4608, 4096)
        self.fc7 = nn.Linear(4096, 4096)
        self.fc8 = nn.Linear(4096, num_classes)

    def forward(self, x):
        x = torch.relu(self.conv1(x))
        x = self.pool1(x)
        x = torch.relu(self.conv2(x))
        x = self.pool2(x)
        x = torch.relu(self.conv3a(x))
        x = torch.relu(self.conv3b(x))
        x = self.pool3(x)
        x = torch.relu(self.conv4a(x))
        x = torch.relu(self.conv4b(x))
        x = self.pool4(x)
        x = torch.relu(self.conv5a(x))
        x = torch.relu(self.conv5b(x))
        x = self.pool5(x)
        x = torch.flatten(x, 1)
        x = torch.relu(self.fc6(x))
        x = F.dropout(x, p=0.5, training=self.training)
        x = torch.relu(self.fc7(x))
        x = F.dropout(x, p=0.5, training=self.training)
        return self.fc8(x)
```

### I3D / SlowFast

```python
class Inception3DModule(nn.Module):
    def __init__(self, in_c, out_1x1, red_3x3, out_3x3, red_5x5, out_5x5, pool_out):
        super().__init__()
        self.branch1 = nn.Conv3d(in_c, out_1x1, kernel_size=1)
        self.branch2 = nn.Sequential(
            nn.Conv3d(in_c, red_3x3, kernel_size=1),
            nn.Conv3d(red_3x3, out_3x3, kernel_size=3, padding=1),
        )
        self.branch3 = nn.Sequential(
            nn.Conv3d(in_c, red_5x5, kernel_size=1),
            nn.Conv3d(red_5x5, out_5x5, kernel_size=3, padding=1),
        )
        self.branch4 = nn.Sequential(
            nn.MaxPool3d(kernel_size=3, stride=1, padding=1),
            nn.Conv3d(in_c, pool_out, kernel_size=1),
        )

    def forward(self, x):
        return torch.cat([
            self.branch1(x), self.branch2(x), self.branch3(x), self.branch4(x)
        ], dim=1)

class SlowFastPathway(nn.Module):
    def __init__(self, num_classes=400):
        super().__init__()
        self.slow_conv1 = nn.Conv3d(3, 64, kernel_size=(1, 7, 7), stride=(1, 2, 2), padding=(0, 3, 3))
        self.fast_conv1 = nn.Conv3d(3, 8, kernel_size=(5, 7, 7), stride=(1, 2, 2), padding=(2, 3, 3))

    def forward(self, x):
        slow = self.slow_conv1(x[:, :, ::2, :, :])
        fast = self.fast_conv1(x)
        return slow, fast
```

### 光流计算实现

```python
import torch
import torch.nn.functional as F

def compute_optical_flow(prev_frame, next_frame):
    prev = prev_frame.float().unsqueeze(0)
    next = next_frame.float().unsqueeze(0)
    grid_y, grid_x = torch.meshgrid(
        torch.arange(prev.shape[2], device=prev.device),
        torch.arange(prev.shape[3], device=prev.device),
        indexing="ij",
    )
    flow = torch.zeros(1, 2, prev.shape[2], prev.shape[3], device=prev.device)
    for i in range(3):
        diff = next - prev
        grad_x = F.conv2d(prev, torch.tensor([[[[-1, 0, 1], [-2, 0, 2], [-1, 0, 1]]]], device=prev.device).float().repeat(1, 3, 1, 1), padding=1)
        grad_y = F.conv2d(prev, torch.tensor([[[[-1, -2, -1], [0, 0, 0], [1, 2, 1]]]], device=prev.device).float().repeat(1, 3, 1, 1), padding=1)
        grad_t = diff
        grad_x = grad_x.mean(dim=1, keepdim=True)
        grad_y = grad_y.mean(dim=1, keepdim=True)
        grad_t = grad_t.mean(dim=1, keepdim=True)
        A = torch.cat([grad_x.unsqueeze(-1), grad_y.unsqueeze(-1)], dim=-1)
        AtA = (A.transpose(-1, -2) @ A).squeeze(-1)
        Atb = -(A.transpose(-1, -2) @ grad_t.unsqueeze(-1)).squeeze(-1)
        flow_update = torch.linalg.solve(
            AtA + torch.eye(2, device=prev.device).view(1, 1, 1, 2, 2) * 0.01,
            Atb.unsqueeze(-1),
        ).squeeze(-1)
        flow = flow + flow_update
        grid = torch.stack([grid_x, grid_y], dim=-1).float() + flow.permute(0, 2, 3, 1)
        grid = grid / torch.tensor([prev.shape[3], prev.shape[2]], device=prev.device).view(1, 1, 1, 2) * 2 - 1
        prev = F.grid_sample(prev_frame.float().unsqueeze(0), grid, mode="bilinear", padding_mode="border", align_corners=False)
    return flow.squeeze(0)
```

### 目标跟踪实现（ByteTrack 风格）

```python
class KalmanFilter:
    def __init__(self):
        self.dt = 1.0
        self.F = torch.tensor([[1, 0, self.dt, 0], [0, 1, 0, self.dt],
                               [0, 0, 1, 0], [0, 0, 0, 1]], dtype=torch.float32)
        self.H = torch.tensor([[1, 0, 0, 0], [0, 1, 0, 0]], dtype=torch.float32)
        self.Q = torch.eye(4) * 0.01
        self.R = torch.eye(2) * 0.1

    def init(self, x, y):
        self.x = torch.tensor([x, y, 0, 0], dtype=torch.float32)
        self.P = torch.eye(4) * 10.0

    def predict(self):
        self.x = self.F @ self.x
        self.P = self.F @ self.P @ self.F.T + self.Q
        return self.x[:2]

    def update(self, zx, zy):
        z = torch.tensor([zx, zy], dtype=torch.float32)
        y = z - self.H @ self.x
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ torch.linalg.inv(S)
        self.x = self.x + K @ y
        self.P = (torch.eye(4) - K @ self.H) @ self.P
        return self.x[:2]

class Track:
    def __init__(self, track_id, x, y):
        self.track_id = track_id
        self.kf = KalmanFilter()
        self.kf.init(x, y)
        self.hits = 1
        self.no_loss = 0

class ByteTracker:
    def __init__(self, det_thresh_high=0.5, det_thresh_low=0.1, track_thresh=0.6):
        self.thresh_high = det_thresh_high
        self.thresh_low = det_thresh_low
        self.track_thresh = track_thresh
        self.tracks = []
        self.next_id = 0

    def update(self, detections):
        high_dets = [d for d in detections if d["score"] >= self.thresh_high]
        low_dets = [d for d in detections if self.thresh_low <= d["score"] < self.thresh_high]
        matched_tracks = []
        unmatched_dets = []

        for track in self.tracks:
            pred = track.kf.predict()
            best_iou, best_det = 0, None
            for det in high_dets:
                iou = self.compute_iou(pred, det["bbox"])
                if iou > best_iou:
                    best_iou, best_det = iou, det
            if best_iou >= self.track_thresh and best_det:
                track.kf.update(best_det["bbox"][0], best_det["bbox"][1])
                track.hits += 1
                track.no_loss = 0
                matched_tracks.append(track)
                high_dets.remove(best_det)
            else:
                track.no_loss += 1
                if track.no_loss < 30:
                    matched_tracks.append(track)

        for det in high_dets:
            track = Track(self.next_id, det["bbox"][0], det["bbox"][1])
            self.next_id += 1
            matched_tracks.append(track)

        self.tracks = matched_tracks
        return self.tracks

    def compute_iou(self, bbox1, bbox2):
        x1, y1 = max(bbox1[0], bbox2[0]), max(bbox1[1], bbox2[1])
        x2, y2 = min(bbox1[2], bbox2[2]), min(bbox1[3], bbox2[3])
        inter = max(0, x2 - x1) * max(0, y2 - y1)
        area1 = (bbox1[2] - bbox1[0]) * (bbox1[3] - bbox1[1])
        area2 = (bbox2[2] - bbox2[0]) * (bbox2[3] - bbox2[1])
        return inter / (area1 + area2 - inter + 1e-6)
```

### 视频数据加载

```python
import torch
import torchvision
from torch.utils.data import Dataset, DataLoader

class VideoDataset(Dataset):
    def __init__(self, video_paths, num_frames=16, frame_size=(224, 224)):
        self.video_paths = video_paths
        self.num_frames = num_frames
        self.frame_size = frame_size

    def __len__(self):
        return len(self.video_paths)

    def __getitem__(self, idx):
        frames = torch.zeros(self.num_frames, 3, *self.frame_size)
        for i in range(min(self.num_frames, 16)):
            frames[i] = torch.rand(3, *self.frame_size)
        label = torch.randint(0, 400, (1,)).item()
        return frames, label

def video_collate(batch):
    frames, labels = zip(*batch)
    return torch.stack(frames), torch.tensor(labels)

video_loader = DataLoader(
    VideoDataset([]), batch_size=8, shuffle=True,
    num_workers=4, collate_fn=video_collate, pin_memory=True,
)
```

```python
model = C3D(num_classes=400).cuda()

for epoch in range(50):
    model.train()
    for videos, labels in video_loader:
        videos, labels = videos.cuda(), labels.cuda()
        videos = videos.permute(0, 2, 1, 3, 4)
        outputs = model(videos)
        loss = criterion(outputs, labels)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
    scheduler.step()

    model.eval()
    correct = total = 0

        for videos, labels in val_loader:
```
