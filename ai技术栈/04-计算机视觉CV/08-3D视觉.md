## 3. 新视角合成

### NeRF（Neural Radiance Fields, 2020）
- **5D 输入**：(x,y,z,θ,φ) → (RGB, σ)
- **体渲染**：沿射线累积颜色和密度
- **问题**：训练慢，渲染慢

### 3D Gaussian Splatting（2023）
- 场景表示为 3D 高斯椭球集合
- **CUDA 快速光栅化**：实时渲染质量达 SOTA

### 新视角合成方法对比

### PointNet 简化实现

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class TNet(nn.Module):
    def __init__(self, k=3):
        super().__init__()
        self.conv1 = nn.Conv1d(k, 64, 1)
        self.conv2 = nn.Conv1d(64, 128, 1)
        self.conv3 = nn.Conv1d(128, 1024, 1)
        self.fc1 = nn.Linear(1024, 512)
        self.fc2 = nn.Linear(512, 256)
        self.fc3 = nn.Linear(256, k * k)

    def forward(self, x):
        batch = x.size(0)
        x = torch.relu(self.conv1(x))
        x = torch.relu(self.conv2(x))
        x = torch.relu(self.conv3(x))
        x = torch.max(x, dim=2)[0]
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        x = self.fc3(x)
        eye = torch.eye(x.size(-1) // 3, device=x.device).view(1, -1).repeat(batch, 1)
        x = x + eye
        return x.view(-1, 3, 3)

class PointNet(nn.Module):
    def __init__(self, num_classes=40):
        super().__init__()
        self.input_tnet = TNet(k=3)
        self.feature_tnet = TNet(k=64)
        self.conv1 = nn.Conv1d(3, 64, 1)
        self.conv2 = nn.Conv1d(64, 64, 1)
        self.conv3 = nn.Conv1d(64, 64, 1)
        self.conv4 = nn.Conv1d(64, 128, 1)
        self.conv5 = nn.Conv1d(128, 1024, 1)
        self.fc1 = nn.Linear(1024, 512)
        self.fc2 = nn.Linear(512, 256)
        self.fc3 = nn.Linear(256, num_classes)
        self.bn1 = nn.BatchNorm1d(64)
        self.bn2 = nn.BatchNorm1d(64)
        self.bn3 = nn.BatchNorm1d(64)
        self.bn4 = nn.BatchNorm1d(128)
        self.bn5 = nn.BatchNorm1d(1024)

    def forward(self, x):
        x = x.transpose(2, 1)
        trans = self.input_tnet(x)
        x = x.transpose(2, 1)
        x = torch.bmm(x, trans)
        x = x.transpose(2, 1)
        x = torch.relu(self.bn1(self.conv1(x)))
        x = torch.relu(self.bn2(self.conv2(x)))
        trans_feat = self.feature_tnet(x)
        x = x.transpose(2, 1)
        x = torch.bmm(x, trans_feat)
        x = x.transpose(2, 1)
        x = torch.relu(self.bn3(self.conv3(x)))
        x = torch.relu(self.bn4(self.conv4(x)))
        x = torch.relu(self.bn5(self.conv5(x)))
        x = torch.max(x, dim=2)[0]
        x = torch.relu(self.fc1(x))
        x = F.dropout(x, p=0.3, training=self.training)
        x = torch.relu(self.fc2(x))
        x = F.dropout(x, p=0.3, training=self.training)
        return self.fc3(x)
```

### NeRF 简化实现

```python
import torch
import torch.nn as nn

class PositionalEncoding(nn.Module):
    def __init__(self, num_freqs=10):
        super().__init__()
        self.num_freqs = num_freqs

    def forward(self, x):
        encoding = [x]
        for i in range(self.num_freqs):
            encoding.append(torch.sin(2.0 ** i * torch.pi * x))
            encoding.append(torch.cos(2.0 ** i * torch.pi * x))
        return torch.cat(encoding, dim=-1)

class NeRF(nn.Module):
    def __init__(self, pos_freqs=10, dir_freqs=4, hidden_dim=256):
        super().__init__()
        self.pos_enc = PositionalEncoding(pos_freqs)
        self.dir_enc = PositionalEncoding(dir_freqs)
        pos_dim = 3 * (2 * pos_freqs + 1)
        dir_dim = 3 * (2 * dir_freqs + 1)

        self.net = nn.Sequential(
            nn.Linear(pos_dim, hidden_dim), nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim), nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim), nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim), nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim), nn.ReLU(),
        )
        self.sigma_head = nn.Linear(hidden_dim, 1)
        self.feature_head = nn.Linear(hidden_dim, hidden_dim)
        self.dir_net = nn.Sequential(
            nn.Linear(hidden_dim + dir_dim, hidden_dim // 2), nn.ReLU(),
            nn.Linear(hidden_dim // 2, 3), nn.Sigmoid(),
        )

    def forward(self, pts, dirs):
        pts_enc = self.pos_enc(pts)
        dirs_enc = self.dir_enc(dirs)
        h = self.net(pts_enc)
        sigma = self.sigma_head(h)
        feat = self.feature_head(h)
        color = self.dir_net(torch.cat([feat, dirs_enc], dim=-1))
        return color, sigma

def volume_rendering(colors, sigmas, z_vals, rays_d):
    dists = z_vals[..., 1:] - z_vals[..., :-1]
    dists = torch.cat([dists, torch.full_like(dists[..., :1], 1e10)], dim=-1)
    alpha = 1 - torch.exp(-sigmas * dists)
    transmittance = torch.cat([
        torch.ones_like(alpha[..., :1]),
        torch.cumprod(1 - alpha + 1e-10, dim=-1)[..., :-1]
    ], dim=-1)
    weights = alpha * transmittance
    rendered_color = torch.sum(weights.unsqueeze(-1) * colors, dim=-2)
    rendered_depth = torch.sum(weights * z_vals, dim=-1)
    return rendered_color, rendered_depth

def sample_rays(H, W, focal, cam_to_world, N_samples=64, near=2.0, far=6.0):
    i, j = torch.meshgrid(torch.arange(W), torch.arange(H), indexing="xy")
    dirs = torch.stack([
        (i - W * 0.5) / focal,
        -(j - H * 0.5) / focal,
        -torch.ones_like(i),
    ], dim=-1)
    rays_d = (dirs @ cam_to_world[:3, :3].T)
    rays_o = cam_to_world[:3, 3].expand_as(rays_d)
    z_vals = torch.linspace(near, far, N_samples).expand(H, W, N_samples)
    return rays_o, rays_d, z_vals

def train_nerf(model, images, poses, focal, device="cuda"):
    optimizer = torch.optim.Adam(model.parameters(), lr=5e-4)
    model.train()
    for epoch in range(10000):
        idx = torch.randint(0, len(images), (1,)).item()
        rays_o, rays_d, z_vals = sample_rays(100, 100, focal, poses[idx], device=device)
        pts = rays_o.unsqueeze(-2) + rays_d.unsqueeze(-2) * z_vals.unsqueeze(-1)
        pts_flat = pts.reshape(-1, 3)
        dirs_flat = rays_d.unsqueeze(-2).expand_as(pts).reshape(-1, 3)
        colors, sigmas = model(pts_flat, dirs_flat)
        colors = colors.reshape(*pts.shape[:2], 3)
        sigmas = sigmas.reshape(*pts.shape[:2], 1)
        rendered, _ = volume_rendering(colors, sigmas, z_vals, rays_d)
        gt = images[idx][:100, :100].to(device)
        loss = F.mse_loss(rendered, gt)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        if epoch % 1000 == 0:
            print(f"Epoch {epoch}: Loss = {loss.item():.6f}")
```

### 3D 点云可视化

```python
import numpy as np
import torch

def generate_sample_point_cloud(num_points=2048):
    theta = torch.rand(num_points) * 2 * torch.pi
    phi = torch.rand(num_points) * torch.pi
    r = 1.0 + 0.1 * torch.randn(num_points)
    x = r * torch.sin(phi) * torch.cos(theta)
    y = r * torch.sin(phi) * torch.sin(theta)
    z = r * torch.cos(phi)
    return torch.stack([x, y, z], dim=1)

def random_rotation(point_cloud, max_angle=torch.pi / 4):
    angle = torch.rand(3) * max_angle
    cos_a, sin_a = torch.cos(angle), torch.sin(angle)
    Rx = torch.tensor([[1, 0, 0], [0, cos_a[0], -sin_a[0]], [0, sin_a[0], cos_a[0]]])
    Ry = torch.tensor([[cos_a[1], 0, sin_a[1]], [0, 1, 0], [-sin_a[1], 0, cos_a[1]]])
    Rz = torch.tensor([[cos_a[2], -sin_a[2], 0], [sin_a[2], cos_a[2], 0], [0, 0, 1]])
    return point_cloud @ (Rx @ Ry @ Rz).T

def farthest_point_sampling(points, n_samples):
    batch, n, _ = points.shape
    centroids = torch.zeros(batch, n_samples, dtype=torch.long, device=points.device)
    distance = torch.ones(batch, n, device=points.device) * 1e10
    farthest = torch.randint(0, n, (batch,), device=points.device)
    for i in range(n_samples):
        centroids[:, i] = farthest
        centroid = points[torch.arange(batch), farthest].unsqueeze(1)
        dist = torch.sum((points - centroid) ** 2, dim=-1)
        mask = dist < distance
        distance[mask] = dist[mask]
        farthest = torch.argmax(distance, dim=-1)
    return centroids

class PointNetPlusPlus(nn.Module):
    def __init__(self, num_classes=40):
        super().__init__()
        self.sa1 = self._set_abstraction(512, 0.2, 32, 3, [64, 64, 128])
        self.sa2 = self._set_abstraction(128, 0.4, 64, 128, [128, 128, 256])
        self.sa3 = self._set_abstraction(None, None, None, 256, [256, 512, 1024])
        self.fc1 = nn.Linear(1024, 512)
        self.fc2 = nn.Linear(512, 256)
        self.fc3 = nn.Linear(256, num_classes)

    def _set_abstraction(self, npoint, radius, nsample, in_c, mlp):
        layers = []
        for c in mlp:
            layers.extend([nn.Conv2d(in_c, c, 1), nn.BatchNorm2d(c), nn.ReLU()])
            in_c = c
        return nn.ModuleDict({
            "npoint": npoint, "radius": radius, "nsample": nsample,
            "mlp": nn.Sequential(*layers),
        })

    def forward(self, xyz):
        return self.fc3(torch.relu(self.fc2(torch.relu(self.fc1(
            torch.max(self.sa3["mlp"](torch.ones(1, 256, 1, 1)), dim=2)[0]
        )))))
```
